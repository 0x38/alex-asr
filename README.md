# Build

    # 1. Checkout PyKaldi
    git checkout https://github.com/UFAL-DSG/pykaldi.git
    
    # 2. Build Pykaldi
    cd pykaldi
    make
    
    # 3. Checkout PyKaldi2
    git checkout https://github.com/UFAL-DSG/pykaldi2.git
    
    # 4. Build
    make py

# Test

    $ make test
